import { writeFileSync } from "node:fs";
import { sql } from "drizzle-orm";
import { db } from "@workspace/db";
import { fetchAirbnbCalendar } from "../../artifacts/api-server/src/lib/ingest/airbnb-calendar-adapter";

type Row = {
  external_id: string;
  date: string;
  nightly_price_usd: number;
  availability_status: "available" | "unavailable" | "unknown";
  minimum_nights: number | null;
  scraped_at: string;
};

function toIsoNow() {
  return new Date().toISOString().replace(/\.\d{3}Z$/, "Z");
}

function dateOnly(d: Date) {
  return d.toISOString().slice(0, 10);
}

function addDays(d: Date, n: number) {
  const x = new Date(d);
  x.setUTCDate(x.getUTCDate() + n);
  return x;
}

function keepDate(d: Date, today: Date): boolean {
  const d90 = addDays(today, 90);
  const d365 = addDays(today, 365);
  if (d < today || d > d365) return false;
  if (d <= d90) return true; // full next 90 days
  // 91..365: weekday+weekend monthly anchors
  const dow = d.getUTCDay(); // 0 Sun ... 6 Sat
  return dow === 1 || dow === 6; // Mon + Sat anchors
}

async function main() {
  const maxListings = Number(process.env.MAX_LISTINGS ?? 2000);
  const outFile = process.env.OUT_FILE ?? "airbnb_calendar_prices.csv";
  const today = new Date();
  const scrapedAt = toIsoNow();

  const result = await db.execute(sql`
    SELECT external_id
    FROM rental_listings
    WHERE source_platform = 'airbnb'
      AND is_active = true
      AND external_id IS NOT NULL
      AND external_id ~ '^[0-9]+$'
    ORDER BY id
    LIMIT ${maxListings}
  `);

  const ids = (result as unknown as { rows: Array<{ external_id: string }> }).rows
    .map(r => r.external_id);

  const rows: Row[] = [];
  let listingsWithAnyPrice = 0;

  for (const externalId of ids) {
    const cal = await fetchAirbnbCalendar(externalId, { monthsCount: 12, timeoutMs: 20000 });

    let listingHadPrice = false;
    for (const day of cal.days) {
      if (day.nightlyPriceUsd == null) continue; // real-only
      const d = new Date(`${day.date}T00:00:00Z`);
      if (!keepDate(d, today)) continue;

      rows.push({
        external_id: externalId,
        date: day.date,
        nightly_price_usd: day.nightlyPriceUsd,
        availability_status: day.availabilityStatus,
        minimum_nights: day.minimumNights,
        scraped_at: scrapedAt,
      });
      listingHadPrice = true;
    }
    if (listingHadPrice) listingsWithAnyPrice++;
  }

  // CSV
  const header = "external_id,date,nightly_price_usd,availability_status,minimum_nights,scraped_at";
  const body = rows.map(r =>
    `${r.external_id},${r.date},${r.nightly_price_usd},${r.availability_status},${r.minimum_nights ?? ""},${r.scraped_at}`
  );
  writeFileSync(outFile, [header, ...body].join("\n"), "utf8");

  console.log(JSON.stringify({
    scanned_listings: ids.length,
    listings_with_any_price: listingsWithAnyPrice,
    rows_written: rows.length,
    out_file: outFile,
    today: dateOnly(today),
  }, null, 2));
}

main().catch((e) => {
  console.error(e);
  process.exit(1);
});
