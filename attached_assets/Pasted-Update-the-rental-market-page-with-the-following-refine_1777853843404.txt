  COUNT(*) FILTER (WHERE availability_status = 'available')::float
    / COUNT(*) AS availability_rate
FROM rental_prices_by_date
WHERE date >= NOW() - INTERVAL '30 days'
GROUP BY date
ORDER BY date;
